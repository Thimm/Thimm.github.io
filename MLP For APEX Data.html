# Helper function to evaluate the total loss on the dataset
def calculate_loss(model, X, y):
    num_examples = len(X)  # training set size
    W1, b1, W2, b2, W3, b3 = model['W1'], model['b1'], \
                             model['W2'], model['b2'], \
                             model['W3'], model['b3']
    # Forward propagation to calculate our predictions
    a1 = forward_prop(X, W1, b1)
    a2 = forward_prop(a1, W2, b2)
    probs = forward_prop_last(a2, W3, b3)
    # Calculating the loss
    corect_logprobs = -np.log(probs[range(num_examples), y])
    data_loss = np.sum(corect_logprobs)
    # Add regulatization term to loss (optional)
    data_loss += reg_lambda / 2 * (np.sum(np.square(W1)) + np.sum(np.square(W2)))
    return 1. / num_examples * data_loss

def predict(model, x):
    W1, b1, W2, b2, W3, b3 = model['W1'], model['b1'], \
                             model['W2'], model['b2'], \
                             model['W3'], model['b3']

    a1 = forward_prop(x, W1, b1)
    a2 = forward_prop(a1, W2, b2)
    probs = forward_prop_last(a2, W3, b3)
    return np.argmax(probs, axis=1)

def calc_accuracy(p, ground_truth):
    return float(np.sum(p == ground_truth)) / float(ground_truth.shape[0])

def initial_layer(input_dim, nodes):
    '''
    initializeds layer based on the input dimensions and on the nr of nodes. 
    input_dim = nr of input signals from the previous layer. 
    nodes = nr of processing units
    '''
    np.random.seed(0)
    W = np.random.randn(input_dim, nodes) / np.sqrt(nn_input_dim)
    b = np.zeros((1, nodes))
    return W, b

def forward_prop(X, W, b):
    ''' calculates values for forward propagation. Returns a matrix calculation of the input as well as 
        the applied tanh.
        X = input to the layer
        W = weights of the layer
        b = bias of the layer 
    '''
    z = X.dot(W) + b # Matrix calc for first layer
    a = np.tanh(z) # appling tanh to it
    return a

def forward_prop_last(X, W, b):
    '''
        Takes the same input than the forward_prop, but calculates the probabilites by squaring the matrix and
        dividing it by the sum
    '''
    z = X.dot(W) + b # Matrix calc for third layer
    exp_scores = np.exp(z) # exponentinal of all values
    return exp_scores / np.sum(exp_scores, axis=1, keepdims=True)

def build_model(X, y, X_t, y_t, num_passes=20000):
    '''
    This function learns parameters for the neural network and returns the model.
    print_loss: If True, print the loss every 1000 iterations
    '''

    # Initialize the parameters to random values. We need to learn these.
    
    # to have the same results every time this is run
    
    W1, b1 = initial_layer(nn_input_dim, nn_hdim1)
    W2, b2 = initial_layer(nn_hdim1, nn_hdim2)
    W3, b3 = initial_layer(nn_hdim2, nn_output_dim)

    # This is what we return at the end
    model = {}
    num_examples = len(X)
    # Gradient descent. For each batch...
    
    for i in range(0, num_passes):
        # Forward propagation. Matrix applications plus aplying the activation funtion
        a1 = forward_prop(X, W1, b1)
        a2 = forward_prop(a1, W2, b2)
        probs = forward_prop_last(a2, W3, b3)

        # Backpropagation
        delta3 = probs
        delta3[range(num_examples), y] -= 1    
        # for final layer
        dW3 = (a2.T).dot(delta3)
        db3 = np.sum(delta3, axis=0, keepdims=True)

        delta2 = delta3.dot(W3.T) * (1 - np.power(a2, 2))
        dW2 = np.dot(a1.T, delta2)
        db2 = np.sum(delta2, axis=0)

        delta1 = delta2.dot(W2.T) * (1 - np.power(a1, 2))
        dW1 = (X.T).dot(delta1)
        db1 = np.sum(delta1, axis=0)

        dW3 += reg_lambda * W3
        # print(dW3.shape, W3.shape, dW2.shape, W2.shape, dW1.shape, W1.shape)
        dW2 += reg_lambda * W2
        dW1 += reg_lambda * W1

        # Gradient descent parameter update
        W1 += -epsilon * dW1
        b1 += -epsilon * db1
        W2 += -epsilon * dW2
        b2 += -epsilon * db2
        W3 += -epsilon * dW3
        b3 += -epsilon * db3

        # Assign new parameters to the model
        model = {'W1': W1, 'b1': b1, 'W2' : W2, 'b2' : b2, 'W3': W3, 'b3': b3}

        # Optionally print the loss.
        # This is expensive because it uses the whole dataset, so we don't want to do it too often.
        if i % 1000 == 0:
            acc = round(calc_accuracy(np.asarray(predict(model, X_t)), y_t), 2)
            print("Loss after iteration %i: %f for training data, %f for testing data. The accuracy for the testing data is %f" % \
                 (i, calculate_loss(model, X, y), calculate_loss(model, X_t, y_t), acc))

    return model
